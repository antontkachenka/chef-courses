nginx Cookbook

This cookbook set up nginx

Platforms

  • Centos 6.7

Chef

  • Chef 12.0 or later

Attributes

nginx::default

Key Type Description
['nginx']['port'] integer port of nginx
['tomcat']['port'] integer port of tomcat
['jenkins']['port'] integer port of jenkins

Usage

Nginx

Just include nginx in your node's run_list:

{
  "name":"my_node",
  "run_list": [
    "recipe[nginx]"
  ]
}

Contributing

  1. Install NGINX via yum install
  2. Replace default.conf file with settings
  3. Run NGINX

package "nginx" do
# Install package NGINX
  action :install
  # RUN ACTION INSTALL
end

template "/etc/nginx/conf.d/default.conf" do
  source "default.erb"
  mode '0755'
  variables ({
    'nginx_port' => node['nginx']['port'],
    'tomcat_port' => node['tomcat']['port'],
    'jenkins_port' => node['jenkins']['port']
})
end

service 'nginx' do
  supports :restart => true, :status => true, :stop => true, :start => true
  action :start
end
